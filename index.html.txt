        function changeLang(lang) {
            currentLang = lang;
            document.querySelectorAll('[data-i18n]').forEach(el => {
                const key = el.getAttribute('data-i18n');
                if (translations[lang][key]) el.innerText = translations[lang][key];
            });

            document.querySelectorAll('.lang-btn').forEach(btn => btn.classList.remove('active'));
            document.getElementById(`btn-${lang}`).classList.add('active');
            renderTools();
            renderFAQ();
        }

        function openModal(name) {
            document.getElementById('modalToolNameDisplay').innerText = name;
            document.getElementById('selectedToolInput').value = name;
            document.getElementById('reserveModal').classList.remove('hidden');
            document.body.style.overflow = 'hidden';
            
            // Užpildome šios dienos datą kaip numatytąją
            const today = new Date().toISOString().split('T')[0];
            document.getElementById('resDate').value = today;
        }

        function closeModal() {
            document.getElementById('reserveModal').classList.add('hidden');
            document.body.style.overflow = 'auto';
        }

        function handleReservation(method) {
            const t = translations[currentLang];
            const toolName = document.getElementById('selectedToolInput').value;
            const date = document.getElementById('resDate').value;
            const phone = document.getElementById('resPhone').value;
            
            const msg = `${t.wa_res_prefix}:\n${t.wa_tool}: ${toolName}\n${t.wa_date}: ${date}${phone ? '\n' + t.wa_phone + ': ' + phone : ''}`;
            
            if (method === 'whatsapp') {
                window.open(`https://wa.me/${myPhoneNumber}?text=${encodeURIComponent(msg)}`, '_blank');
            } else if (method === 'viber') {
                window.open(`viber://chat?number=${myPhoneNumber}&draft=${encodeURIComponent(msg)}`, '_blank');
                // Fallback jei viber:// neveikia tiesiogiai
                setTimeout(() => {
                    window.location.href = `viber://pa?chatURI=${myPhoneNumber}&text=${encodeURIComponent(msg)}`;
                }, 100);
            }
            closeModal();
        }

        function sendGeneralWhatsApp() {
            window.open(`https://wa.me/${myPhoneNumber}?text=Sveiki, domina nuoma.`, '_blank');
        }

        window.onclick = (e) => { if (e.target == document.getElementById('reserveModal')) closeModal(); }
